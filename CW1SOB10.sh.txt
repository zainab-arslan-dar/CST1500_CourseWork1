#!/bin/bash

# Check if the number of arguments is exactly 3
if [ "$#" -ne 3 ]; then
  echo "Usage: $0 <path> <x> <y>"
  exit 1
fi

# Assign the arguments to variables
path="$1"
x="$2"
y="$3"

# Check if the path exists, if not create it
if [ -d "$path" ]; then
  echo "The directory '$path' already exists."
else
  echo "The directory '$path' does not exist. Creating it now..."
  mkdir -p "$path"
  echo "Directory '$path' created successfully."
fi
